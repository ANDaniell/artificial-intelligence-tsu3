```
import json  
import os  
import torch  
from transformers import AutoTokenizer, AutoModelForCausalLM, pipeline, BitsAndBytesConfig  
from langchain_huggingface.llms import HuggingFacePipeline  
from langchain.text_splitter import RecursiveCharacterTextSplitter  
from langchain_huggingface.embeddings import HuggingFaceEmbeddings  # –ú–æ–¥–µ–ª—å —ç–º–±–µ–¥–¥–∏–Ω–≥–æ–≤  
from langchain_community.vectorstores import FAISS  # –í–µ–∫—Ç–æ—Ä–Ω–æ–µ —Ö—Ä–∞–Ω–∏–ª–∏—â–µ FAISS  
from langchain.chains import RetrievalQA  # –¶–µ–ø–æ—á–∫–∞ –¥–ª—è –ø–æ–∏—Å–∫–∞ –∏ –æ—Ç–≤–µ—Ç–∞  
  
book_path = "book.txt"  
index_path = "faiss_index"  
model_name = "mistralai/Mistral-7B-Instruct-v0.2"  
  
  
# –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ–≥–æ FAISS-–∏–Ω–¥–µ–∫—Å–∞  
def is_faiss_index_valid(path):  
    return (  
            os.path.isdir(path) and  
            os.path.exists(os.path.join(path, "index.faiss")) and  
            any(fname.endswith(".pkl") or fname.endswith(".json") for fname in os.listdir(path))  
    )  
  
  
# –ü—Ä–µ–¥–æ–±—Ä–∞–±–æ—Ç–∫–∞ —Ç–µ–∫—Å—Ç–∞ –∏ –ø–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ –∏–Ω–¥–µ–∫—Å–∞ (–µ—Å–ª–∏ –æ–Ω –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç)  
def preprocess_book(book_path=book_path, index_path=index_path):  
    # –ó–∞–≥—Ä—É–∑–∫–∞ —Ç–µ–∫—Å—Ç–∞  
    with open(book_path, "r", encoding="utf-8") as f:  
        text = f.read()  
  
    # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –Ω–∞ —Ñ—Ä–∞–≥–º–µ–Ω—Ç—ã (—á–∞–Ω–∫–∏)  
    text_splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=100)  
    docs = text_splitter.create_documents([text])  
  
    # –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –∏–Ω–¥–µ–∫—Å–æ–≤ —á–∞–Ω–∫–æ–≤ –≤ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã–µ  
    for i, doc in enumerate(docs):  
        doc.metadata["chunk_index"] = i  
  
    # –°–æ–∑–¥–∞–Ω–∏–µ —ç–º–±–µ–¥–¥–∏–Ω–≥–æ–≤ –∏ –∏–Ω–¥–µ–∫—Å  
    embedding_model = HuggingFaceEmbeddings(model_name="sentence-transformers/all-MiniLM-L6-v2")  
    vectorstore = FAISS.from_documents(docs, embedding_model)  
    vectorstore.save_local(index_path)  
    return vectorstore, embedding_model  
  
  
# –ó–∞–≥—Ä—É–∑–∫–∞ —è–∑—ã–∫–æ–≤–æ–π –º–æ–¥–µ–ª–∏  
def load_model(model_name=model_name):  
    device = "cuda" if torch.cuda.is_available() else "cpu"  
    # –ö–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è –¥–ª—è –∑–∞–≥—Ä—É–∑–∫–∏ –º–æ–¥–µ–ª–∏ –≤ 4-–±–∏—Ç–Ω–æ–º —Ñ–æ—Ä–º–∞—Ç–µ –¥–ª—è RTX 4060  
    bnb_config = BitsAndBytesConfig(  
        load_in_4bit=True,  
        bnb_4bit_use_double_quant=True,  
        bnb_4bit_quant_type="nf4",  
        bnb_4bit_compute_dtype=torch.float16  
    )  
    # –ó–∞–≥—Ä—É–∑–∫–∞ –º–æ–¥–µ–ª–∏ —Å —É—á–µ—Ç–æ–º —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞  
    if device == "cuda":  
        model = AutoModelForCausalLM.from_pretrained(  
            model_name,  
            quantization_config=bnb_config,  
            device_map="auto"  
        )  
    else:  
        model = AutoModelForCausalLM.from_pretrained(model_name)  
  
    tokenizer = AutoTokenizer.from_pretrained(model_name)  
    # –û–±–µ—Ä—Ç–∫–∞ pipeline Hugging Face  
    hf_pipeline = pipeline(  
        task="text-generation",  
        model=model,  
        tokenizer=tokenizer,  
        temperature=0.3,  # –ú–µ–Ω—å—à–µ ‚Äî –¥–µ—Ç–µ—Ä–º–∏–Ω–∏—Ä–æ–≤–∞–Ω–Ω–µ–µ  
        max_new_tokens=528,  
        return_full_text=False,  
        do_sample=True,  
        pad_token_id=tokenizer.eos_token_id  
    )  
  
    llm = HuggingFacePipeline(pipeline=hf_pipeline)  
    return llm  
  
  
# –ó–∞–ø—É—Å–∫ —Ü–µ–ø–æ—á–∫–∏ QA –Ω–∞ —Å–ø–∏—Å–∫–µ –≤–æ–ø—Ä–æ—Å–æ–≤  
def run_qa(qa_chain, questions):  
    results = []  
    for q in questions:  
        response = qa_chain.invoke(q)  
        results.append({  
            "question": q,  
            "answer": response["result"],  
            "sources": [doc.metadata for doc in response["source_documents"]]  
        })  
        print(f"Q: {q}\nA: {response['result']}\n")  
    return results  
  
  
if __name__ == "__main__":  
    # –ü—Ä–µ–¥–æ–±—Ä–∞–±–æ—Ç–∫–∞ (–µ—Å–ª–∏ –∏–Ω–¥–µ–∫—Å –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç)  
    if not is_faiss_index_valid(index_path):  
        print("–°–æ–∑–¥–∞–Ω–∏–µ –Ω–æ–≤–æ–≥–æ –∏–Ω–¥–µ–∫—Å–∞...")  
        vectorstore, embedding_model = preprocess_book()  
    else:  
        print("–ò–Ω–¥–µ–∫—Å –Ω–∞–π–¥–µ–Ω. –ó–∞–≥—Ä—É–∂–∞–µ–º...")  
        embedding_model = HuggingFaceEmbeddings(model_name="sentence-transformers/all-MiniLM-L6-v2")  
        vectorstore = FAISS.load_local(index_path, embedding_model, allow_dangerous_deserialization=True)  
    # –ó–∞–≥—Ä—É–∑–∫–∞ –º–æ–¥–µ–ª–∏  
    llm = load_model()  
  
    # –ü–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ RAG —Ü–µ–ø–æ—á–∫–∏  
    qa = RetrievalQA.from_chain_type(  
        llm=llm,  
        retriever=vectorstore.as_retriever(search_type="similarity", search_kwargs={"k": 3}),  
        return_source_documents=True  
    )  
  
    # –í–æ–ø—Ä–æ—Å—ã –¥–ª—è —Ç–µ—Å—Ç–∞  
    questions = [  
        "What is Kant's definition of pure reason?",  
        "How does Kant distinguish between a priori and a posteriori knowledge?",  
        "What are the main categories of understanding according to Kant?",  
        "What is the role of transcendental aesthetics in Kant's philosophy?",  
        "How does Kant argue for the limits of metaphysics?"  
    ]  
  
    # –ó–∞–ø—É—Å–∫ —Ç–µ—Å—Ç–æ–≤ –∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤  
    results = run_qa(qa, questions)  
    with open("qa_results.json", "w", encoding="utf-8") as f:  
        json.dump(results, f, ensure_ascii=False, indent=2)
```
